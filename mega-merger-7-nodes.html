def count_districts_in_city_name(name):
    replacables = [("[", ""), ("]", ""), ("A", "_"), ("F", "_")]
    for replacable in replacables:
        name = name.replace(*replacable)
    ids = set()
    for node_id in name.split("_"):
        ids.add(int(node_id))
    return len(ids)
